Customize list style using child render function:

<Canvas>
  <Story name="Customized List Style">
    {() => {
      return (

          name="selected-list"

            <Box mb={2}>
              <Box as="span" mr={2}>
                {radio}
              </Box>{' '}

          )}

      );
